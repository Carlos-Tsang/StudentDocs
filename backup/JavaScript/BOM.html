BOM 浏览器对象模型

navigator

平台信息，可获取客户端的设备信息

    //Navigator对象
    var a = navigator.appCodeName,
        //浏览器代码名
        b = navigator.appName,
        //浏览器名称
        c = navigator.appVersion,
        //平台及版本信息
        d = navigator.cookieEnabled,
        //cookie是否被启用,返回布尔值
        e = navigator.platform,
        //系统平台信息
        f = navigator.userAgent;
        //返回客户端的user-agent头部的值

screen

屏幕信息，可获取客户端的屏幕宽高

    //screen对象
    var a1 = screen.availHeight,
        //返回显示屏幕的可用高度
        a2 = screen.availWidth,
        //返回显示屏幕的可用宽度
        a3 = screen.height,
        //返回屏幕的像素高度
        a4 = screen.width,
        //返回屏幕的像素宽度
        a5 = screen.colorDepth;
        //返回屏幕的颜色位数

history(重点)

历史信息，可进行前进、后退操作。

    //history对象
        //history.back();//后退一页
        //history.forward();//前进一页
        //history.go(1);//正数即为前进，负数即为后退
    var b1 = history.length;
        //返回历史记录的条数

location(重点)

位置信息，可获取url成分，进行跳转操作。

    //location对象
    var c1 = location.hash,
        //返回锚点包括#的字符串
        c2 = location.host,
        //主机名 + 端口号
        c3 = location.hostname,
        //返回主机名
        c4 = location.port,
        //返回端口号
        c5 = location.href,
        //返回完整的URL
        //location.href = 'math.html';
        c6 = location.protocol,
        //返回当前URL的协议
        c7 = location.search;
        //返回包括?的搜索参数
        //location.assign('math.html');//当前页面加载新文档
        //location.reload();
        //location.replace('math.html');
        //以新文档替换当前文档。

窗口操作(慎用)

非IE浏览器存在问题。

    myWindow = window.open('','','width=200,height=100');
    //创建新窗口
    myWindow.document.write("This is 'myWindow'");
    myWindow.moveTo(200,200);
    //移动窗口到...位置
    myWindow.moveBy(100,0);
    //移动窗口到相对现在的...位置
    myWindow.resizeBy(2, 1);
    //重置窗口大小
    myWindow.close();
    //关闭新窗口

    console.log('IE:' + screenLeft);//IE,窗口左上角的X轴坐标

文档滚动距离

窗口滚动距离，可放在scroll事件中，Top<=>Left，X<=>Y

    window.onscroll = function(){
        console.log('IE:' + document.documentElement.scrollTop);
    }

    window.onscroll = function(){
        console.log('~:' + pageYOffset);
    }

    var wTop = screenTop || screenY;//IE与其他浏览器有区别
